# Set the minimum version of CMake that can be used
# To find the cmake version run
# $ cmake --version
cmake_minimum_required(VERSION 3.5)

# Set the project name
project (mysql_connector)

# Create a sources variable with a link to all cpp files to compile
set(SOURCES
    src/main.cpp
)

link_directories($ENV{MYSQL_CPPCONN_DIR}/lib64/vs14
$ENV{BOOST_LIB_DIR})


# Add an executable with the above sources
add_executable(mysql_connector ${SOURCES})

# Set the directories that should be included in the build command for this target
# when running g++ these will be included as -I/directory/path/
target_include_directories(mysql_connector
    PRIVATE 
        ${PROJECT_SOURCE_DIR}/include
        #$ENV{MYSQL_CPPCONN_DIR}/include
        $ENV{MYSQL_CPPCONN_JBDC_DIR}
        $ENV{BOOSTDIR}
)

target_link_libraries(mysql_connector  
    #mysqlcppconn8.lib
    #mysqlcppconn8-static.lib
    #ssleay32.lib
    mysqlcppconn.lib
    )
    
if (MSVC)
set(CompilerFlags
    CMAKE_CXX_FLAGS
    CMAKE_CXX_FLAGS_DEBUG
    CMAKE_CXX_FLAGS_RELEASE
    CMAKE_C_FLAGS
    CMAKE_C_FLAGS_DEBUG
    CMAKE_C_FLAGS_RELEASE
    )
foreach(CompilerFlag ${CompilerFlags})
    string(REPLACE "/MD" "/MT" ${CompilerFlag} "${${CompilerFlag}}")
endforeach()
endif(MSVC)

if(MSVC)
  # Force to always compile with W4
  if(CMAKE_CXX_FLAGS MATCHES "/W[0-4]")
    string(REGEX REPLACE "/W[0-4]" "/W4" CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS}")
  else()
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /W4")
  endif()
elseif(CMAKE_COMPILER_IS_GNUCC OR CMAKE_COMPILER_IS_GNUCXX)
  # Update if necessary
  set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wno-long-long -pedantic")
endif()
